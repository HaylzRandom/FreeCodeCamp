  let alphabetsGroup = [
   {
   	'?':'?',

   	'.' : '.',
  ' ': ' ',
   	A: "N",
   	B: "O",
   	C: "P",
   	D: "Q",
   	E: "R",
   	F: "S",
   	G: "T",

   	I: "V",
   	J: "W",
   	K: "X",

   	M: "Z",
   	N: "A",
   	O: "B",
   	P: "C",

   	R: "E",
   	S: "F",

   	U: "H",
   	V: "I",
   	W: "J",
   	X: "K",
   	Y: "L",
   	Z: "M"
   }
  ];
  let newWord = [];

	for(let i = 0; i < str.length; i++){
			newWord.push(alphabetsGroup[0][str[i]])
	}
	return newWord.join('')
}
